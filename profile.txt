Set-PSReadLineKeyHandler -Key "RightArrow" -ScriptBlock {
       param($key, $arg)

       $line = $null
       $cursor = $null
       [Microsoft.PowerShell.PSConsoleReadLine]::GetBufferState([ref]$line, [ref]$cursor)

       if ($cursor -lt $line.Length) {
           [Microsoft.PowerShell.PSConsoleReadLine]::ForwardChar($key, $arg)
       } else {
           [Microsoft.PowerShell.PSConsoleReadLine]::AcceptNextSuggestionWord($key, $arg)
       }
}

Set-PSReadLineKeyHandler -Key End -ScriptBlock {
       param($key, $arg)

       $line = $null
       $cursor = $null
       [Microsoft.PowerShell.PSConsoleReadLine]::GetBufferState([ref]$line, [ref]$cursor)

       if ($cursor -lt $line.Length) {
           [Microsoft.PowerShell.PSConsoleReadLine]::EndOfLine($key, $arg)
       } else {
           [Microsoft.PowerShell.PSConsoleReadLine]::AcceptSuggestion($key, $arg)
       }
}

function Kill-Port {
    param(
        [Parameter(Mandatory=$true)]
        [int]$Port
    )
    $pid = (Get-NetTCPConnection -LocalPort $Port -ErrorAction Stop).OwningProcess[0]
    taskkill /F /PID $pid
}

# --- BONSAI Sync ---
$BONSAI_REPO = "C:\Git\BONSAI"

function Invoke-BonsaiPush {
    $global = "$env:USERPROFILE\.claude"
    foreach ($item in @('output-styles', 'rules', 'skills')) {
        Copy-Item -Path "$BONSAI_REPO\.claude\$item" -Destination $global -Recurse -Force
    }
    Copy-Item -Path "$BONSAI_REPO\CLAUDE.md" -Destination "$env:USERPROFILE\CLAUDE.md" -Force
    Write-Host "BONSAI pushed to global"
}
Set-Alias -Name bonsai-push -Value Invoke-BonsaiPush

function Invoke-BonsaiPull {
    $global = "$env:USERPROFILE\.claude"
    foreach ($item in @('output-styles', 'rules', 'skills')) {
        Copy-Item -Path "$global\$item" -Destination "$BONSAI_REPO\.claude" -Recurse -Force
    }
    Copy-Item -Path "$env:USERPROFILE\CLAUDE.md" -Destination "$BONSAI_REPO\CLAUDE.md" -Force
    Write-Host "BONSAI pulled from global"
}
Set-Alias -Name bonsai-pull -Value Invoke-BonsaiPull

function Invoke-BonsaiDiff {
    $pairs = @(
        @{ Name = 'output-styles'; G = "$env:USERPROFILE\.claude\output-styles"; R = "$BONSAI_REPO\.claude\output-styles" }
        @{ Name = 'rules';         G = "$env:USERPROFILE\.claude\rules";         R = "$BONSAI_REPO\.claude\rules" }
        @{ Name = 'skills';        G = "$env:USERPROFILE\.claude\skills";        R = "$BONSAI_REPO\.claude\skills" }
        @{ Name = 'CLAUDE.md';     G = "$env:USERPROFILE\CLAUDE.md";             R = "$BONSAI_REPO\CLAUDE.md" }
    )

    Write-Host "BONSAI sync status " -NoNewline
    Write-Host "(global <-> BONSAI repo)" -ForegroundColor DarkGray
    Write-Host ""

    $details = @()
    foreach ($p in $pairs) {
        $name = $p.Name; $g = $p.G; $r = $p.R
        if (-not (Test-Path $g) -or -not (Test-Path $r)) {
            $side = if (-not (Test-Path $g)) { 'global' } else { 'repo' }
            Write-Host ("  {0,-14} " -f $name) -NoNewline
            Write-Host "! missing " -ForegroundColor Yellow -NoNewline
            Write-Host "($side side not found)" -ForegroundColor DarkGray
            continue
        }

        $diffs = @()
        if (Test-Path $g -PathType Container) {
            $gMap = @{}; $rMap = @{}
            Get-ChildItem -Path $g -Recurse -File | ForEach-Object {
                $gMap[$_.FullName.Substring($g.Length).TrimStart('\')] = (Get-FileHash $_.FullName -Algorithm SHA256).Hash
            }
            Get-ChildItem -Path $r -Recurse -File | ForEach-Object {
                $rMap[$_.FullName.Substring($r.Length).TrimStart('\')] = (Get-FileHash $_.FullName -Algorithm SHA256).Hash
            }
            foreach ($k in $gMap.Keys) {
                if (-not $rMap.ContainsKey($k))   { $diffs += "Only in global: $k" }
                elseif ($gMap[$k] -ne $rMap[$k])  { $diffs += "Differ:         $k" }
            }
            foreach ($k in $rMap.Keys) {
                if (-not $gMap.ContainsKey($k))   { $diffs += "Only in repo:   $k" }
            }
            $gMtime = (Get-ChildItem -Path $g -Recurse -File | Sort-Object LastWriteTime | Select-Object -Last 1).LastWriteTime
            $rMtime = (Get-ChildItem -Path $r -Recurse -File | Sort-Object LastWriteTime | Select-Object -Last 1).LastWriteTime
        } else {
            if ((Get-FileHash $g -Algorithm SHA256).Hash -ne (Get-FileHash $r -Algorithm SHA256).Hash) {
                $diffs += "Content differs"
            }
            $gMtime = (Get-Item $g).LastWriteTime
            $rMtime = (Get-Item $r).LastWriteTime
        }

        Write-Host ("  {0,-14} " -f $name) -NoNewline
        if ($diffs.Count -eq 0) {
            Write-Host "OK in sync" -ForegroundColor Green
            continue
        }
        if ($gMtime -gt $rMtime) {
            Write-Host "-> global newer " -ForegroundColor Cyan -NoNewline
            Write-Host "(run: bonsai-pull) " -ForegroundColor DarkGray -NoNewline
        } elseif ($rMtime -gt $gMtime) {
            Write-Host "<- repo newer   " -ForegroundColor Cyan -NoNewline
            Write-Host "(run: bonsai-push) " -ForegroundColor DarkGray -NoNewline
        } else {
            Write-Host "! same mtime, content differs " -ForegroundColor Yellow -NoNewline
        }
        Write-Host ("-- {0} file(s) differ" -f $diffs.Count) -ForegroundColor DarkGray
        $details += [PSCustomObject]@{ Name = $name; Diffs = $diffs }
    }

    if ($details.Count -gt 0) {
        Write-Host ""
        Write-Host "Details"
        foreach ($d in $details) {
            Write-Host ""
            Write-Host $d.Name
            $d.Diffs | ForEach-Object { Write-Host "  $_" }
        }
    }
}
Set-Alias -Name bonsai-diff -Value Invoke-BonsaiDiff
